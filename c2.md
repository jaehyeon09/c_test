#include <stdio.h>

struct Student {
    char name[20];
    int korean;
    int english;
    int math;
};

int main() {
    struct Student students[2] = {
        {"Minsu", 90, 85, 95},
        {"Jisoo", 80, 75, 85}
    };

    double studentAvg;
    double koreanAvg, englishAvg, mathAvg;

    // 학생 평균 출력
    printf("===== 학생 평균 =====\n");

    for (int i = 0; i < 2; i++) {
        studentAvg = (students[i].korean +
                      students[i].english +
                      students[i].math) / 3.0;

        printf("%s의 평균 : %.2f\n",
               students[i].name, studentAvg);
    }

    // 과목 평균 계산
    koreanAvg = (students[0].korean + students[1].korean) / 2.0;
    englishAvg = (students[0].english + students[1].english) / 2.0;
    mathAvg = (students[0].math + students[1].math) / 2.0;

    // 과목 평균 출력
    printf("\n===== 과목 평균 =====\n");
    printf("국어 평균 : %.2f\n", koreanAvg);
    printf("영어 평균 : %.2f\n", englishAvg);
    printf("수학 평균 : %.2f\n", mathAvg);

    return 0;
}
