```
#include "raylib.h"

#define WIDTH 800
#define HEIGHT 600

int main(void)
{
    InitWindow(WIDTH, HEIGHT, "Rainbow Block Breaker");
    SetTargetFPS(60);

    Color rainbow[7] = { RED, ORANGE, YELLOW, GREEN, BLUE, DARKBLUE, VIOLET };

    Rectangle paddle = { 350, 550, 100, 20 };
    Vector2 ball = { 400, 300 };
    Vector2 speed = { 4, -4 };

    Rectangle blocks[7];
    bool active[7];

    for (int i = 0; i < 7; i++)
    {
        blocks[i] = (Rectangle){ 50 + i * 100, 100, 80, 30 };
        active[i] = true;
    }

    int score = 0;
    int life = 5;

    while (!WindowShouldClose() && life > 0)
    {
        // 패들 이동
        if (IsKeyDown(KEY_LEFT))
            paddle.x -= 6;

        if (IsKeyDown(KEY_RIGHT))
            paddle.x += 6;

        // 공 이동
        ball.x += speed.x;
        ball.y += speed.y;

        // 벽 충돌
        if (ball.x <= 10 || ball.x >= WIDTH - 10)
            speed.x *= -1;

        if (ball.y <= 10)
            speed.y *= -1;

        // 패들 충돌
        if (CheckCollisionCircleRec(ball, 10, paddle))
            speed.y = -fabs(speed.y);

        // 블록 충돌
        for (int i = 0; i < 7; i++)
        {
            if (active[i] &&
                CheckCollisionCircleRec(ball, 10, blocks[i]))
            {
                active[i] = false;
                speed.y *= -1;
                score += 10;
            }
        }

        // 공이 바닥으로 떨어짐
        if (ball.y > HEIGHT)
        {
            life--;

            ball.x = 400;
            ball.y = 300;

            speed.x = 4;
            speed.y = -4;

            paddle.x = 350;
        }

        BeginDrawing();
        ClearBackground(RAYWHITE);

        DrawText(TextFormat("Score : %d", score), 20, 20, 30, BLACK);
        DrawText(TextFormat("Life : %d", life), 20, 60, 30, BLACK);

        // 무지개 블록
        for (int i = 0; i < 7; i++)
        {
            if (active[i])
                DrawRectangleRec(blocks[i], rainbow[i]);
        }

        DrawRectangleRec(paddle, BLACK);
        DrawCircleV(ball, 10, MAROON);

        EndDrawing();
    }

    // 게임 오버 화면
    while (!WindowShouldClose())
    {
        BeginDrawing();
        ClearBackground(BLACK);

        DrawText("GAME OVER", 250, 250, 50, RED);
        DrawText(TextFormat("FINAL SCORE : %d", score),
            220, 330, 30, WHITE);

        EndDrawing();
    }

    CloseWindow();
    return 0;
}
```
