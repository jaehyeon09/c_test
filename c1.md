### ←, → 방향키를 이용하여 패들을 좌우로 이동합니다.
### 공이 블록에 맞으면 블록이 제거되고 점수(Score)가 10점씩 증가합니다.
### 공이 바닥 아래로 떨어지면 실행 횟수(Play)가 1 증가하며 공과 패들이 초기 위치로 돌아갑니다.
### 총 5회까지 플레이 가능하며, 5회가 지나면 게임이 종료됩니다.
### 화면 왼쪽 위에서 현재 점수(Score) 와 실행 횟수(Play) 를 확인할 수 있습니다.
```
#include "raylib.h"
#include <math.h>

#define WIDTH 800
#define HEIGHT 600

int main(void)
{
    InitWindow(WIDTH, HEIGHT, "Rainbow Block Breaker");
    SetTargetFPS(60);

    Color rainbow[7] = { RED, ORANGE, YELLOW, GREEN, BLUE, DARKBLUE, VIOLET };

    Rectangle blocks[7];
    bool active[7];

    for (int i = 0; i < 7; i++)
    {
        blocks[i] = (Rectangle){ 50 + i * 100, 100, 80, 30 };
        active[i] = true;
    }

    Rectangle paddle = { 350, 550, 100, 20 };
    Vector2 ball = { 400, 300 };
    Vector2 speed = { 4, -4 };

    int score = 0;
    int playCount = 1;   // 현재 시도 횟수

    while (!WindowShouldClose() && playCount <= 5)
    {
        // 패들 이동
        if (IsKeyDown(KEY_LEFT)) paddle.x -= 6;
        if (IsKeyDown(KEY_RIGHT)) paddle.x += 6;

        // 공 이동
        ball.x += speed.x;
        ball.y += speed.y;

        // 벽 충돌
        if (ball.x <= 10 || ball.x >= WIDTH - 10) speed.x *= -1;
        if (ball.y <= 10) speed.y *= -1;

        // 패들 충돌
        if (CheckCollisionCircleRec(ball, 10, paddle))
            speed.y = -fabs(speed.y);

        // 블록 충돌
        for (int i = 0; i < 7; i++)
        {
            if (active[i] && CheckCollisionCircleRec(ball, 10, blocks[i]))
            {
                active[i] = false;
                speed.y *= -1;
                score += 10;
            }
        }

        // 공이 바닥에 떨어짐
        if (ball.y > HEIGHT)
        {
            playCount++;

            // 5회 초과면 종료
            if (playCount > 5) break;

            // 공과 패들 위치만 초기화
            ball = (Vector2){ 400, 300 };
            speed = (Vector2){ 4, -4 };
            paddle.x = 350;
        }

        BeginDrawing();
        ClearBackground(RAYWHITE);

        DrawText(TextFormat("Score : %d", score), 20, 20, 30, BLACK);
        DrawText(TextFormat("Play : %d / 5", playCount), 20, 60, 30, BLACK);

        for (int i = 0; i < 7; i++)
        {
            if (active[i])
                DrawRectangleRec(blocks[i], rainbow[i]);
        }

        DrawRectangleRec(paddle, BLACK);
        DrawCircleV(ball, 10, MAROON);

        EndDrawing();
    }

    // 종료 화면
    while (!WindowShouldClose())
    {
        BeginDrawing();
        ClearBackground(BLACK);

        DrawText("GAME OVER", 250, 250, 50, RED);
        DrawText(TextFormat("FINAL SCORE : %d", score),
            220, 330, 30, WHITE);

        EndDrawing();
    }

    CloseWindow();
    return 0;
}
```
